## 代码示例模板

```python
# 模块示例：my_module.py
"""
这是一个示例模块
"""

def hello():
    """模块中的函数"""
    return "Hello from my_module!"

class MyClass:
    """模块中的类"""
    def __init__(self, name):
        self.name = name
    
    def greet(self):
        return f"Hello, {self.name}!"

# 模块变量
VERSION = "1.0.0"

# 当模块被直接运行时执行
if __name__ == "__main__":
    print(hello())
    obj = MyClass("World")
    print(obj.greet())

# 导入示例
# 方式1：导入整个模块
import my_module
print(my_module.hello())

# 方式2：导入特定函数或类
from my_module import hello, MyClass
print(hello())

# 方式3：导入并重命名
import my_module as mm
print(mm.hello())

# 方式4：导入所有内容（不推荐）
from my_module import *
print(hello())

# 包结构示例：
# my_package/
# ├── __init__.py
# ├── module1.py
# ├── module2.py
# └── subpackage/
#     ├── __init__.py
#     └── module3.py

# 在包中使用相对导入
# 在 my_package/module1.py 中：
# from . import module2  # 导入同级模块
# from .subpackage import module3  # 导入子包中的模块
```
