## 代码示例模板

```python
import time
from datetime import datetime, date, time, timedelta, timezone
import pytz

# time 模块
print(f"当前时间戳: {time.time()}")
print(f"本地时间元组: {time.localtime()}")
print(f"格式化时间: {time.strftime('%Y-%m-%d %H:%M:%S')}")

# 程序计时
start_time = time.time()
time.sleep(1)  # 延时1秒
end_time = time.time()
print(f"执行时间: {end_time - start_time:.2f}秒")

# datetime 模块
# date 对象
today = date.today()
print(f"今天: {today}")
print(f"年: {today.year}, 月: {today.month}, 日: {today.day}")

# time 对象
now_time = time(14, 30, 45)
print(f"时间: {now_time}")

# datetime 对象
now = datetime.now()
print(f"当前日期时间: {now}")
print(f"格式化: {now.strftime('%Y-%m-%d %H:%M:%S')}")

# timedelta 对象
delta = timedelta(days=7, hours=2)
future = now + delta
print(f"一周后: {future}")

# 时间计算
birthday = date(1990, 5, 15)
age = today - birthday
print(f"天数: {age.days}")
print(f"年数: {age.days // 365}")

# 时间解析
date_str = "2023-05-15 14:30:00"
parsed = datetime.strptime(date_str, "%Y-%m-%d %H:%M:%S")
print(f"解析结果: {parsed}")

# 时区处理
utc_now = datetime.now(timezone.utc)
beijing_tz = pytz.timezone('Asia/Shanghai')
beijing_now = utc_now.astimezone(beijing_tz)

print(f"UTC时间: {utc_now}")
print(f"北京时间: {beijing_now}")

# 工作日计算
def add_workdays(start_date, days):
    """添加工作日"""
    result = start_date
    added = 0
    while added < days:
        result += timedelta(days=1)
        if result.weekday() < 5:  # 周一到周五
            added += 1
    return result

start = date(2023, 5, 15)
end = add_workdays(start, 10)
print(f"10个工作日后: {end}")

# 时间工具函数
def is_weekend(date_obj):
    """判断是否是周末"""
    return date_obj.weekday() >= 5

def get_month_start(date_obj):
    """获取月初"""
    return date_obj.replace(day=1)

def get_month_end(date_obj):
    """获取月末"""
    next_month = date_obj.replace(day=28) + timedelta(days=4)
    return next_month - timedelta(days=next_month.day)

print(f"今天是否周末: {is_weekend(today)}")
print(f"本月第一天: {get_month_start(today)}")
print(f"本月最后一天: {get_month_end(today)}")
```
