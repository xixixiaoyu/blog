## 代码示例模板

```python
import matplotlib.pyplot as plt
import seaborn as sns
import plotly.express as px
import numpy as np
import pandas as pd

# Matplotlib 示例
def matplotlib_example():
    # 创建数据
    x = np.linspace(0, 10, 100)
    y1 = np.sin(x)
    y2 = np.cos(x)
    
    # 创建图形
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 4))
    
    # 折线图
    ax1.plot(x, y1, label='sin(x)')
    ax1.plot(x, y2, label='cos(x)')
    ax1.set_title('三角函数')
    ax1.legend()
    ax1.grid(True)
    
    # 散点图
    x_scatter = np.random.randn(100)
    y_scatter = np.random.randn(100)
    ax2.scatter(x_scatter, y_scatter, alpha=0.6)
    ax2.set_title('散点图')
    ax2.set_xlabel('X')
    ax2.set_ylabel('Y')
    
    plt.tight_layout()
    plt.show()

# Seaborn 示例
def seaborn_example():
    # 创建示例数据
    tips = sns.load_dataset("tips")
    
    # 创建子图
    fig, axes = plt.subplots(2, 2, figsize=(12, 10))
    
    # 箱线图
    sns.boxplot(x='day', y='total_bill', data=tips, ax=axes[0,0])
    axes[0,0].set_title('箱线图')
    
    # 小提琴图
    sns.violinplot(x='day', y='total_bill', data=tips, ax=axes[0,1])
    axes[0,1].set_title('小提琴图')
    
    # 热力图
    corr = tips[['total_bill', 'tip', 'size']].corr()
    sns.heatmap(corr, annot=True, cmap='coolwarm', ax=axes[1,0])
    axes[1,0].set_title('相关性热力图')
    
    # 分布图
    sns.histplot(tips['total_bill'], kde=True, ax=axes[1,1])
    axes[1,1].set_title('分布图')
    
    plt.tight_layout()
    plt.show()

# Plotly 示例
def plotly_example():
    # 创建示例数据
    df = px.data.iris()
    
    # 散点图
    fig1 = px.scatter(df, x='sepal_width', y='sepal_length', 
                     color='species', size='petal_length',
                     title='鸢尾花数据集')
    fig1.show()
    
    # 3D 散点图
    fig2 = px.scatter_3d(df, x='sepal_length', y='sepal_width', 
                        z='petal_length', color='species',
                        title='3D 鸢尾花数据')
    fig2.show()
    
    # 动画散点图
    fig3 = px.scatter(df, x='sepal_width', y='sepal_length',
                     animation_frame='species', color='species',
                     title='动画散点图')
    fig3.show()

if __name__ == "__main__":
    print("Matplotlib 示例:")
    matplotlib_example()
    
    print("Seaborn 示例:")
    seaborn_example()
    
    print("Plotly 示例:")
    plotly_example()
```
