## 代码示例模板

```python
import numpy as np

# 数组创建
arr1 = np.array([1, 2, 3, 4, 5])
arr2 = np.zeros((3, 3))
arr3 = np.ones((2, 4))
arr4 = np.random.rand(3, 3)
arr5 = np.arange(0, 10, 2)
arr6 = np.linspace(0, 1, 5)

# 数组属性
print(f"数组形状: {arr1.shape}")
print(f"数组维度: {arr1.ndim}")
print(f"数据类型: {arr1.dtype}")
print(f"数组大小: {arr1.size}")

# 数组操作
# 索引和切片
print(f"arr1[0]: {arr1[0]}")
print(f"arr1[1:4]: {arr1[1:4]}")

# 形状操作
reshaped = arr1.reshape(5, 1)
flattened = reshaped.flatten()

# 数组运算
a = np.array([1, 2, 3])
b = np.array([4, 5, 6])
print(f"加法: {a + b}")
print(f"乘法: {a * b}")
print(f"点积: {np.dot(a, b)}")

# 聚合函数
print(f"求和: {arr1.sum()}")
print(f"平均值: {arr1.mean()}")
print(f"最大值: {arr1.max()}")
print(f"最小值: {arr1.min()}")

# 文件操作
np.save('array.npy', arr1)
loaded_arr = np.load('array.npy')
np.savetxt('array.txt', arr1)
```
