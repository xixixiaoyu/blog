## 代码示例模板

```python
import time
import datetime
import calendar

# time 模块
timestamp = time.time()
print(f"当前时间戳: {timestamp}")
print(f"格式化时间: {time.strftime('%Y-%m-%d %H:%M:%S')}")
print(f"性能测量: {time.perf_counter()}")

# datetime 模块
now = datetime.datetime.now()
today = datetime.date.today()
current_time = datetime.time(12, 30, 45)

# 日期运算
future_date = today + datetime.timedelta(days=30)
past_date = today - datetime.timedelta(days=7)

# 时区处理
import pytz
utc_time = datetime.datetime.now(pytz.UTC)
beijing_time = utc_time.astimezone(pytz.timezone('Asia/Shanghai'))

# calendar 模块
print(f"2024年是否闰年: {calendar.isleap(2024)}")
print(f"2024年2月天数: {calendar.monthrange(2024, 2)[1]}")
```
