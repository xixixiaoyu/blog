## 代码示例模板

```python
import math
import decimal
import fractions
import random

# 基本数字类型
integer_num = 42
float_num = 3.14159
complex_num = 2 + 3j

# math 模块使用
print(f"π = {math.pi}")
print(f"e = {math.e}")
print(f"sin(π/2) = {math.sin(math.pi/2)}")
print(f"log(10) = {math.log(10)}")

# decimal 高精度计算
decimal.getcontext().prec = 10
a = decimal.Decimal('0.1')
b = decimal.Decimal('0.2')
print(f"0.1 + 0.2 = {a + b}")  # 精确的 0.3

# fractions 分数运算
frac1 = fractions.Fraction(1, 3)
frac2 = fractions.Fraction(2, 3)
print(f"1/3 + 2/3 = {frac1 + frac2}")

# random 模块
random.seed(42)  # 设置随机种子
print(f"随机整数: {random.randint(1, 10)}")
print(f"随机浮点数: {random.random()}")
choices = ['apple', 'banana', 'orange']
print(f"随机选择: {random.choice(choices)}")
```
